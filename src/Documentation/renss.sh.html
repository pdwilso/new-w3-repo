#!/bin/bash
	for fn in `ls -Q Screenshot\ 2023*.png | tr ' ' '_'` ; do\
  	  ifn=$fn ;;
	  echo \$MV $ifn\
	    `echo $fn | \
	     sed -e 's/Screenshot_202/sc/' | \
	     sed -e 's/[-_.]//g' | \
	     sed -e 's/png/\.png/' `  ;;
	done

Produces output like 

	$MV "Screenshot_2023-06-14_14.49.36.png" "sc30614144936.png"
	$MV  "Screenshot_2023-06-14_20.35.53.png" "sc30614203553.png"
	$MV  "Screenshot_2023-06-14_20.36.06.png" "sc30614203606.png"
	$MV  "Screenshot_2023-06-15_20.54.54.png" "sc30615205454.png"
	$MV  "Screenshot_2023-06-15_21.00.24.png" "sc30615210024.png"
	$MV  "Screenshot_2023-06-16_01.22.37.png" "sc30616012237.png"
